1) Analysis

EDA → segmentation → cohorts → driver analysis.

Reusable cohort SQL (template)

-- Example cohort pattern (anonymized)
WITH first_event AS (
  SELECT user_id, MIN(event_date) AS cohort_date
  FROM events
  GROUP BY 1
),
activity AS (
  SELECT e.user_id, f.cohort_date, e.event_date
  FROM events e
  JOIN first_event f USING(user_id)
)
SELECT cohort_date,
       DATE_DIFF(event_date, cohort_date, DAY) AS day_n,
       COUNT(DISTINCT user_id) AS users
FROM activity
GROUP BY 1,2
ORDER BY 1,2;
